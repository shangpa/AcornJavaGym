## 전체 클래스 다이어그램

```mermaid
classDiagram
    %% ─────────────── 인터페이스 ───────────────
    class Registerable {
        <<interface>>
        +register() void
    }

    class Payable {
        <<interface>>
        +pay(months: int) void
    }

    class DietManageable {
        <<interface>>
        +manageDiet(member: Member) void
    }

    class FitnessEquipmentGuide {
        <<interface>>
        +explainEquipment(bodyPart: String) void
    }

    %% ─────────────── 열거형 ───────────────
    class TrainerGrade {
        <<enumeration>>
        INTERN
        JUNIOR
        SENIOR
        MASTER
    }

    %% ─────────────── 추상 클래스 ───────────────
    class Person {
        <<abstract>>
        -name: String
        -phoneNumber: String
        -age: int
        -height: double
        -weight: double
        +Person(name, phoneNumber, age, height, weight)
        +getBMI(height: double, weight: double) double
        +getBMICategory() void
        +printInfo()* void
        +getName() String
        +setName(name: String) void
        +getPhoneNumber() String
        +setPhoneNumber(phoneNumber: String) void
        +getAge() int
        +setAge(age: int) void
        +getHeight() double
        +setHeight(height: double) void
        +getWeight() double
        +setWeight(weight: double) void
    }

    %% ─────────────── MembershipType ───────────────
    class MembershipType {
        +ONE_MONTH: MembershipType$
        +THREE_MONTH: MembershipType$
        +SIX_MONTH: MembershipType$
        +TWELVE_MONTHS: MembershipType$
        -months: final int
        -price: final int
        +MembershipType(months: int, price: int)
        +getMonths() int
        +getPrice() int
    }

    %% ─────────────── Member ───────────────
    class Member {
        -registrationDate: LocalDate
        -expirationDate: LocalDate
        -expirationDate_after_extension: LocalDate
        -attendedToday: boolean
        -membership: MembershipType
        -formatter: DateTimeFormatter
        +Member(name, phoneNumber, age, height, weight)
        +register() void
        +pay(months: int) void
        +explainEquipment(bodyPart: String) void
        +checkIn() void
        +printInfo() void
        +setMembership(membership: MembershipType) void
        +getMembership() MembershipType
        +getExpirationDate() LocalDate
        +isAttendedToday() boolean
    }

    %% ─────────────── Trainer ───────────────
    class Trainer {
        -grade: TrainerGrade
        -managedMembers: ArrayList~Member~
        +Trainer(name, phoneNumber, age, height, weight)
        +register() void
        +manageDiet(member: Member) void
        +addManagedMember(member: Member) void
        +printManagedMembers() void
        +recommendWorkout(member: Member) void
        -updateGradeByMemberCount() void
        +printAttendance() void
        +printInfo() void
        +getGrade() TrainerGrade
        +getManagedMembers() ArrayList~Member~
    }

    %% ─────────────── Locker ───────────────
    class Locker {
        -lockerNumber: int
        -isOccupied: boolean
        -assignedMember: Member
        +Locker(lockerNumber: int)
        +assign(member: Member) void
        +release() void
        +toGridCell() String
        +printInfo() void
        +getLockerNumber() int
        +isOccupied() boolean
        +getAssignedMember() Member
    }

    %% ─────────────── GymManager ───────────────
    class GymManager {
        -memberList: ArrayList~Member~
        -trainerList: ArrayList~Trainer~
        -lockerList: ArrayList~Locker~
        +addMember(member: Member) void
        +addTrainer(trainer: Trainer) void
        +addLocker(locker: Locker) void
        +findMember(name: String) Member
        +findTrainer(name: String) Trainer
        +findLocker(lockerNumber: int) Locker
        +printAllMembers() void
        +printAllTrainers() void
        +printAllLocker() void
        +printLockerGrid() void
        +getMemberList() ArrayList~Member~
        +getTrainerList() ArrayList~Trainer~
        +getLockerList() ArrayList~Locker~
        +setLockerList(lockerList: ArrayList~Locker~) void
    }

    %% ─────────────── Main ───────────────
    class Main {
        -sc: Scanner$
        -gm: GymManager$
        +main(args: String[])$ void
        -memberMenu()$ void
        -trainerMenu()$ void
        -equipmentMenu()$ void
        -lockerMenu()$ void
    }

    %% ─────────────── 관계 ───────────────
    Person <|-- Member : extends
    Person <|-- Trainer : extends

    Member ..|> Registerable : implements
    Member ..|> Payable : implements
    Member ..|> FitnessEquipmentGuide : implements

    Trainer ..|> Registerable : implements
    Trainer ..|> DietManageable : implements

    Member --> MembershipType : uses
    Trainer --> TrainerGrade : uses
    Trainer "1" o-- "0..*" Member : manages

    Locker --> Member : assignedMember

    GymManager "1" o-- "0..*" Member : manages
    GymManager "1" o-- "0..*" Trainer : manages
    GymManager "1" o-- "0..*" Locker : manages

    Main --> GymManager : uses
```

## 상속 / 구현 관계 다이어그램

    Person <|-- Trainer

## 객체 연관 관계 다이어그램

```mermaid
classDiagram
    direction LR

    class GymManager {
        -memberList: ArrayList~Member~
        -trainerList: ArrayList~Trainer~
        -lockerList: ArrayList~Locker~
    }
    class Trainer {
        -managedMembers: ArrayList~Member~
        -grade: TrainerGrade
    }
    class Member {
        -membership: MembershipType
    }
    class Locker {
        -lockerNumber: int
        -isOccupied: boolean
        -assignedMember: Member
    }
    class MembershipType {
        +ONE_MONTH$
        +THREE_MONTH$
        +SIX_MONTH$
        +TWELVE_MONTHS$
    }
    class TrainerGrade {
        <<enumeration>>
        INTERN
        JUNIOR
        SENIOR
        MASTER
    }

    GymManager "1" o-- "0..*" Member : memberList
    GymManager "1" o-- "0..*" Trainer : trainerList
    GymManager "1" o-- "0..*" Locker : lockerList
    Trainer "1" o-- "0..*" Member : managedMembers
    Trainer --> TrainerGrade : grade
    Member --> MembershipType : membership
    Locker --> Member : assignedMember
```

## 관계 범례

| 기호 | 의미 |
|------|------|
| `<\|--` | 상속 (extends) |
| `..\|>` | 구현 (implements) |
| `o--` | 집합 연관 (has-a, ArrayList) |
| `-->` | 단방향 연관 (uses) |
| `<<abstract>>` | 추상 클래스 |
| `<<interface>>` | 인터페이스 |
| `<<enumeration>>` | 열거형 |
| `$` | static 멤버 |
| `*` | abstract 메서드 |
